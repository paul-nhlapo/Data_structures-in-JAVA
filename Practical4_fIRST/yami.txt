public class Main {
    public static void main(String[] args) {
        minDHeap<Integer> heap = new minDHeap<>(3);

        for (int i = 10; i >= 0; i--) {
            heap.insert(i);
        }
        System.out.println(heap);
        heap.changeD(3);
        System.out.println(heap);
        System.out.println("\nPath to root for index 5:");
        System.out.println(heap.pathToRoot(5));
        System.out.println(heap);
        System.out.println("\nPath to root for index 7:");
        System.out.println(heap.pathToRoot(7));
        heap.changeD(2);
        System.out.println("\nThe new heap after changing the d value:");
        System.out.println(heap);
        System.out.println("\nPath to root for index 5:");
        System.out.println(heap.pathToRoot(5));
        System.out.println(heap);
        System.out.println("\nPath to root for index 7:");
        System.out.println(heap.pathToRoot(7));
        for (int i = 0; i <= 5; i++) {
            heap.remove(i);
        }
        System.out.println(heap);

        // Now test the pathToRoot function for index 3
        System.out.println("\nPath to root for index 7:");
        System.out.println(heap);
        System.out.println(heap.pathToRoot(7));

    }

}
